### ❓ Q3：再次评估增加20万，能合并计算吗？

**结论：用交易流水制（Transaction Journal），自然就能合并。**

#### 🌍 全球参考案例 — 成本计算公式

**Ghostfolio 的做法（加权平均成本法）**：

```
多次买入同一基金的成本合并：

BUY #1: 10万 买入，净值 1.2，份额 = 83,333
BUY #2: 20万 买入，净值 1.5，份额 = 133,333
───────────────────────────────────
合计：30万 成本，216,666 份额
加权平均成本净值 = 300,000 / 216,666 = ¥1.385

当前净值 1.6 时：
市值 = 216,666 × 1.6 = ¥346,666
总盈亏 = ¥346,666 - ¥300,000 = +¥46,666 (+15.6%)
```

**基金平台的通用做法**（支付宝/天天基金）：
```
持仓成本 = (第1次买入金额 + 第2次买入金额 + ... + 手续费) / 总份额
```

### ❓ Q4：卖掉了，持仓怎么更新？

**结论：引入 SELL 交易类型，自动更新持仓。**

#### 🌍 全球参考案例

| App | 卖出机制 |
|-----|---------|
| **Ghostfolio** | SELL activity 记录卖出份额+价格 → 自动减持仓、算已实现盈亏 |
| **Monarch** | 交易列表中记录卖出 → 持仓自动减少 |
| **Empower** | 连接券商自动同步卖出 → 也支持手动记录 |
| **支付宝** | 赎回确认后自动更新持仓，保留历史交易记录 |
| **天天基金** | 确认赎回 → 份额减少 → 成本不变 → 显示已实现收益 |

### 关键计算函数

```javascript
/**
 * 从交易流水计算当前持仓（加权平均成本法）
 * 参考：Ghostfolio Portfolio Calculator
 */
function calcHoldings(transactions) {
  const holdings = {};  // code → { shares, totalCost, avgNav }
  const realized = {};  // code → 已实现盈亏

  for (const tx of transactions.sort((a,b) => new Date(a.date) - new Date(b.date))) {
    if (!holdings[tx.code]) {
      holdings[tx.code] = { code: tx.code, name: tx.name, shares: 0, totalCost: 0, avgNav: 0, txCount: 0 };
    }
    const h = holdings[tx.code];

    if (tx.type === 'BUY') {
      h.totalCost += tx.amount + (tx.fee || 0);
      h.shares += tx.shares;
      h.avgNav = h.totalCost / h.shares;  // 加权平均
      h.txCount++;
    } 
    else if (tx.type === 'SELL') {
      const sellCost = tx.shares * h.avgNav;        // 卖出部分的成本
      const sellRevenue = tx.shares * tx.nav - (tx.fee || 0);  // 卖出收入
      realized[tx.code] = (realized[tx.code] || 0) + (sellRevenue - sellCost);
      h.totalCost -= sellCost;
      h.shares -= tx.shares;
      // avgNav 不变（加权平均成本法特性）
      if (h.shares <= 0) h.shares = 0;
    }
  }

  // 过滤掉已清仓的
  return {
    active: Object.values(holdings).filter(h => h.shares > 0),
    realized,  // 已实现盈亏
    closed: Object.values(holdings).filter(h => h.shares <= 0),
  };
}

/**
 * 净资产 = 投资市值 + 现金类资产 + 固定资产 - 负债
 */
function calcNetWorth(holdings, currentNavs, assets) {
  let investmentValue = 0;
  for (const h of holdings) {
    investmentValue += h.shares * (currentNavs[h.code] || h.avgNav);
  }
  
  const cashTotal = assets.filter(a => a.type === 'cash').reduce((s, a) => s + a.balance, 0);
  const propertyTotal = assets.filter(a => a.type === 'property').reduce((s, a) => s + a.value, 0);
  const otherTotal = assets.filter(a => a.type === 'other').reduce((s, a) => s + a.value, 0);
  const liabilityTotal = assets.filter(a => a.type === 'liability').reduce((s, a) => s + Math.abs(a.balance), 0);
  
  return {
    total: investmentValue + cashTotal + propertyTotal + otherTotal - liabilityTotal,
    breakdown: { investment: investmentValue, cash: cashTotal, property: propertyTotal, other: otherTotal, liability: liabilityTotal }
  };
}
```

## ⚠️ 迁移策略（不丢老数据）

```javascript
// 旧 → 新 自动迁移
function migrateV3toV4(oldPortfolio) {
  const transactions = [];
  
  for (const h of oldPortfolio.holdings) {
    transactions.push({
      id: 'migrate_' + h.code,
      type: 'BUY',
      code: h.code,
      name: h.name,
      amount: h.amount,
      shares: 0,  // 旧数据没有份额，后端补算
      nav: 0,     // 需要后端根据 buyDate 查历史净值补填
      fee: 0,
      date: h.buyDate,
      source: 'recommend',
      note: 'V3迁移',
    });
  }
  
  return {
    transactions,
    assets: [],
    profile: oldPortfolio.profile,
    version: 4,
  };
}
```
